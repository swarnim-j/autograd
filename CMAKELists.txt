cmake_minimum_required(VERSION 3.10)

# Project Name
project(autograd)

# C++ Standard
set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED True)

# Add this line to specify the include directories
include_directories(src)

# Source files
set(SOURCES
    src/main.cpp
)

# Header files
set(HEADERS
    src/Tensor.h
    src/AutogradOps.h
    src/operations/Operation.h
    src/operations/Mul.h
    src/operations/Add.h
    src/operations/ReLU.h
    src/operations/Sigmoid.h
    src/operations/Softmax.h
    src/operations/Tanh.h
)

# Add executable
add_executable(autograd ${SOURCES} ${HEADERS})
